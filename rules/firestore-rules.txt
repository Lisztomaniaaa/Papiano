// ══════════════════════════════════════════════════════════════════════
// FIRESTORE SECURITY RULES — Papiano (Locked: server-first approach)
// Deploy: Firebase Console → Firestore → Rules → Paste → Publish
//
// PHILOSOPHY: Client can READ. All WRITES go through server API
// (Vercel Functions with Firebase Admin SDK bypass these rules).
// Only minimal client writes are kept for UX (reactions, read markers).
// ══════════════════════════════════════════════════════════════════════

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    // ─── HELPER FUNCTIONS ───
    function isAuth() {
      return request.auth != null;
    }
    function isOwner(uid) {
      return isAuth() && request.auth.uid == uid;
    }

    // ─── PROFILES ───
    match /profiles/{uid} {
      // Anyone authenticated can read profiles
      allow read: if isAuth();

      // Client can ONLY create their own profile on first sign-in
      // (ensureUserProfile in app). All updates go through /api/profile.
      allow create: if isOwner(uid)
        && request.resource.data.name is string
        && request.resource.data.name.size() <= 24;

      // Client can update ONLY safe display fields (no ownedRoles, no playTimeSeconds, no banned)
      // Full updates go through /api/profile server endpoint.
      allow update: if isOwner(uid)
        && (!('ownedRoles' in request.resource.data) || request.resource.data.ownedRoles == resource.data.ownedRoles)
        && (!('playTimeSeconds' in request.resource.data) || request.resource.data.playTimeSeconds == resource.data.playTimeSeconds)
        && (!('banned' in request.resource.data) || request.resource.data.banned == resource.data.banned)
        && (!('isBanned' in request.resource.data) || request.resource.data.isBanned == resource.data.isBanned);

      // Owner can delete their own profile
      allow delete: if isOwner(uid);

      // Reactions sub-collection (likes/dislikes) — keep client-writable for UX
      match /reactions/{voterId} {
        allow read: if isAuth();
        allow write: if isAuth() && request.auth.uid == voterId;
        allow delete: if isAuth() && request.auth.uid == voterId;
      }
    }

    // ─── COUNTERS (auto-increment publicId) ───
    match /counters/{counterId} {
      allow read: if isAuth();
      allow write: if isAuth()
        && request.resource.data.next is number
        && request.resource.data.next > 0;
    }

    // ─── DONATIONS (server-only write) ───
    match /donations/{donationId} {
      allow read: if isAuth();
      allow write: if false; // Admin SDK only via /api/admin
    }

    // ─── FRIENDSHIPS (server-only write via /api/friends) ───
    match /friendships/{pairId} {
      // Both users can read
      allow read: if isAuth()
        && (request.auth.uid in resource.data.users
            || request.auth.uid == resource.data.receiverId
            || request.auth.uid == resource.data.requesterId);

      // ALL writes go through /api/friends
      allow create: if false;
      allow update: if false;
      allow delete: if false;
    }

    // ─── BLOCKS (server-only write via /api/moderate) ───
    match /blocks/{blockId} {
      allow read: if isAuth()
        && (request.auth.uid == resource.data.blockerId || request.auth.uid == resource.data.blockedId);

      // ALL writes go through /api/moderate
      allow create: if false;
      allow update: if false;
      allow delete: if false;
    }

    // ─── CHAT ROOMS ───
    match /chatRooms/{roomId} {
      // Participants can read
      allow read: if isAuth()
        && (resource.data.type == 'group'
            || request.auth.uid in resource.data.participants);

      // Room metadata updates (read markers, unread count) — keep client-writable
      allow create: if isAuth();
      allow update: if isAuth()
        && (resource.data.type == 'group'
            || request.auth.uid in resource.data.participants);

      // Messages sub-collection — server-only write via /api/chat
      match /messages/{messageId} {
        allow read: if isAuth();

        // ALL message writes go through /api/chat
        allow create: if false;
        allow update: if false;
        allow delete: if false;
      }
    }

    // ─── REPORTS (server-only write via /api/moderate) ───
    match /reports/{reportId} {
      allow read: if isAuth() && request.auth.uid == resource.data.reporterId;

      // ALL writes go through /api/moderate
      allow create: if false;
      allow update: if false;
      allow delete: if false;
    }

    // ─── CATCH-ALL: deny everything else ───
    match /{document=**} {
      allow read, write: if false;
    }
  }
}
